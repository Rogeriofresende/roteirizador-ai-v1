```typescript
interface FormValidationRule {
  type: 'required' | 'minLength' | 'maxLength' | 'pattern' | 'email' | 'custom' | 'crossField';
  field?: string;
  dependsOn?: string[];
  validator?: (value: any, formData?: Record<string, any>) => boolean | Promise<boolean>;
  async?: boolean;
  message: string;
}
```
